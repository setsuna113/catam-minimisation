\label{tab:q1-minima}
\begin{tabular}{lll}
\toprule
Function & Minimiser \((x^{*},y^{*})\) & \(f^{*}\)\\
\midrule
(4) & \((-0.3700,\,-0.7937)\) & \(-1.0953\)\\
(5) & \((1,\,1)\) & \(0\)\\
\bottomrule
\end{tabular}

\textbf{Function (4).}
\begin{itemize}
\item Solve \(\partial_x f = 1+x-y^{2} = 0\):\ \ \(x = y^{2}-1\).
\item Sub into \(\partial_y f = 1-2y+4y(y^{2}-x/2) = 0\):\ \ \(1+2y^{3}=0 \Rightarrow y^{*} = -2^{-1/3},\ \ x^{*} = 2^{-2/3}-1\) (unique real root).
\item Hessian: \(H = \bigl(\begin{smallmatrix}1 & -2y^{*}\\ -2y^{*} & -2+12y^{*2}-2x^{*}\end{smallmatrix}\bigr) \approx \bigl(\begin{smallmatrix}1 & 1.587\\ 1.587 & 6.300\end{smallmatrix}\bigr)\);\ \ \(\det H \approx 3.78 > 0,\ \operatorname{tr} H > 0\) \(\Rightarrow\) local min.
\item \((y^{2}-x/2)^{2} \to \infty\) as \(|(x,y)| \to \infty\) \(\Rightarrow\) also global min.
\item Eliminate \(x=y^{2}-1\) in \(f\):\ \ \(f^{*} = -\tfrac{1}{2} - 2^{-1/3} + 2^{-7/3} \approx -1.0953\).
\end{itemize}

\textbf{Function (5).}
\begin{itemize}
\item \(f = (1-x)^{2} + 80(y-x^{2})^{2} \ge 0\), with equality iff \(y = x^{2}\) and \(x = 1\) \(\Rightarrow\) \((x^{*},y^{*}) = (1,1),\ f^{*} = 0\), unique global min.
\item Hessian at \((1,1)\): \(H = \bigl(\begin{smallmatrix}642 & -320\\ -320 & 160\end{smallmatrix}\bigr)\);\ \ \(\det H = 320 > 0,\ \operatorname{tr} H > 0\) \(\Rightarrow\) strict min confirmed.
\end{itemize}
